\documentclass[11pt,Chicago]{styles/uuthesis2e}


% load packages
\usepackage{styles/packages}


% better math support (and undefine two macros from uu-thesis)
\let \proof    = \relax
\let \endproof = \relax
\usepackage{amsmath}
\usepackage{amsfonts}


% font for dissertation
% default Computer Modern is replaced, DejaVuSerifCondensed and fourier nice too
\usepackage{libertine}


% define levels (chapter is top-most, then section and sub-section)
\fourlevels


% additional uu-thesis styles (override settings from above packages!)
\usepackage{styles/uuthesis-2016-h}


% add colored headings
\usepackage{styles/uuthesis-color-headings}
\definecolor{utahheadingcolor}{rgb}{0.1, 0.1, 0.1}


% load custom styles & commands
\usepackage{styles/custom}


% load bibliography & journal abbreviations file
\addbibresource{$journals$}
\addbibresource{$references$}


% use double-spaces between references
\setlength\bibitemsep{2\itemsep}


% front matter
\author{$author$}
\title{$title$}
\thesistype{dissertation}


% degree & department info
\degree{$degree$\\ in\\ $degreeIn$}
\approvaldepartment{$degreeIn$}
\department{$department$}
\graduatedean{$graduateDean$}
\departmentchair{$departmentChair$}


% committee & dates
\committeechair{$advisor$}
\firstreader{$committee1$}
\secondreader{$committee2$}
\thirdreader{$committee3$}
\fourthreader{$committee4$}
\chairtitle{Professor}
\submitdate{$submitted$ $year$}
\copyrightyear{$year$}
\chairdateapproved{$day$ $month$ $year$}
\firstdateapproved{$day$ $month$ $year$}
\seconddateapproved{$day$ $month$ $year$}
\thirddateapproved{$day$ $month$ $year$}
\fourthdateapproved{$day$ $month$ $year$}


\begin{document}


% front matter content
\frontmatterformat
\titlepage
\copyrightpage
\dissertationapproval
\setcounter{page}{2} % abstract on p. iii
\preface{$abstract$}{Abstract}
\toc
\preface{$acknowledge$}{Acknowledgments}


% chapters
\maintext
\pagestyle{headings}
$for(chapter)$
  \include{$chapter$}
$endfor$


% appendices
\numberofappendices = $appendixCount$
\appendix
$for(appendix)$
  \include{$appendix$}
$endfor$


% references (from uuthesis2e.cls)
\newpage
  \thispagestyle{empty}%
  \addcontentsline{toc}{chapter}{REFERENCES}%
  \mainheading{REFERENCES}%
  \par\removelastskip\singlespace\par\removelastskip% GBG Oct 1993
